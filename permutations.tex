\section{Permutation}


\begin{itemize}

\item Suppose we have a set of \textbf{n} items.
\item From that set, we select \textbf{k} items.
\item The number of \textbf{ordered} lists of \textbf{k} items chosen from a set of \textbf{n} items is

\[\frac{n!}{n-k!}\]
\end{itemize}
%================================%

\[ {n \choose r} = \frac{n!}{(n-r)! r!} \]


\[ {6 \choose 3} = \frac{6!}{(6-3)! 3!} = \frac{6!}{3! \times 3!}\]


\[ \frac{6!}{3! \times 3!} = \frac{6 \times 5 \times 4 \times 3!}{3! \times 3!} = \frac{120}{6} = 120\]
\end{itemize}
