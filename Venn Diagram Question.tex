\begin{frame}
\textbf{A college teaches courses in mathematics, computing and statistics,and students can choose a range of courses from these three subject areas. \\
\bigskip
Currently 600 students are enrolled of whom 300 study mathematics courses, 120 study statistics
and 380 study computing courses. }

\end{frame}
%-----------------------------------------------------------%
\begin{frame}

\textbf{Let the subject areas be represented by the letters M for mathematics, C for computing and S for statistics. 
\\
\bigskip
Draw a labelled Venn diagram showing the areas M, C, and S in such a way as to represent the students studying at
the college. 
\\
\bigskip
On your diagram show the number of students studying in each region of the Venn diagram.
}
\end{frame}
%-----------------------------------------------------------%
\begin{frame}
\textbf{
\begin{itemize}
\item 40 students study courses from all three subject areas. 
\bigskip
\item 200 mathematics students study computing subjects as well.
\bigskip 
\item 60 computing students also study statistics subjects.
\bigskip
\item 70 statistics students also study mathematics subjects.
\end{itemize}
}
\end{frame}

%-----------------------------------------------------------%
\begin{frame}
\textbf{
Questions:
\begin{itemize}
\item How many students study none of these courses at all? \bigskip \bigskip
\item How many students study mathematics but not computing or statistics? \bigskip \bigskip
\item How many students study courses from precisely two of these subject areas?
\end{itemize}
}
